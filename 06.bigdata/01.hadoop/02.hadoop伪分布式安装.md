##拷贝 hadoop-2.6.2.tar.gz到~/app
tar -zxf hadoop-2.6.2.tar.gz

##配置hadoop环境变量
	vim ~/.bashrc
	HADOOP_HOME=/home/hadoop/app/hadoop-2.6.2
	PATH=$PATH:$HADOOP_HOME/bin
	export HADOOP_HOME
